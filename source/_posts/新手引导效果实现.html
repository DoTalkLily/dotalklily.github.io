新手效果在很多网站中都能看到，很好地引导用户了解网站相关操作从而降低学习成本。本文的新手引导效果基于 intro.js 做了修改，介绍下引入方式，然后讲解下网站中常看到的右下角能展开收齐的帮助弹框实现方式。

引入 

<script type="text/javascript" src="http://g.alicdn.com/base-tech-fe/base_fe_common/src/common/controller/intro.js"></script>                      
<link href="http://g.alicdn.com/base-tech-fe/base_fe_common/src/common/styles/introjs.css" rel="stylesheet"> 

注：插件用intro.js

html

  • 在需要显示引导的元素里添加标识属性，如
<div data-step="1"></div>

js

  • 添加js
function startIntro(){
          var intro = introJs();
            intro.setOptions({
              steps: [
                {
                  element: '[data-step="1"]',
                  intro: "这里是数据引擎提供的开发套件，如果当前区域为灰色，说明您选择的工作台模板暂时不需要该功能。 您可以在右上角的“设置-设置工作台”中开启相应的模块。",
                  position: 'bottom'
                },
                {
                  element: '[data-step="2"]',
                  intro: "在这里可以管理项目的数据、数据服务、数据源和数据主题。",
                  position: 'right'
                },
                {
                  element: '[data-step="3"]',
                  intro: '在这里可以为项目添加成员，管理成员权限，配置项目空间。',
                  position: 'right'
                }
              ],
              'nextLabel': "下一步",                                           
              'prevLabel': "上一步",
              'skipLabel': "跳过",
              'doneLabel':"完成",
              'showStepNumbers':false
            });
            intro.start();
                   

tips添加方式

引入 

<script type="text/javascript" src="http://g.alicdn.com/base-tech-fe/base_fe_common/src/common/controller/tooltip.js"></script>

html 

<div class="toolTip" data-tooltip title="tenant1"></div>

js 

var keyMap = {
    'tenant1': '提供一站式的数据集成开发环境，包括数据查询、ETL开发、数据算法挖掘等功能。'
};
$('[data-tooltip]').tooltip({verticalOffset:-200, horizontalOffset:20, keyMap: keyMap});
